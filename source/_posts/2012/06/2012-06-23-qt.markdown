### Qt x11 - ubuntu 12.04安装依赖 ###

	sudo apt-get install -y libgl1-mesa-dev libglu1-mesa-dev libice-dev libiodbc2-dev libjpeg-dev libmng-dev

	sudo apt-get install -y libtiff4-dev
